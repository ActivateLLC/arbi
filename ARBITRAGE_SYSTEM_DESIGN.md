### Risk Scoring Algorithm

```typescript
function scoreOpportunity(opp: Opportunity): Score {
  return {
    // 0-100 scoring
    profitPotential: calculateROI(opp) * 10,
    confidence: historicalSuccessRate(opp.type) * 100,
    timeToProfit: (30 - daysToSell) / 30 * 100,
    liquidity: marketDepth * 10,

    // Risk factors (negative points)
    volatility: -priceSwings * 20,
    complexity: -executionSteps * 5,

    // Final decision
    shouldExecute: totalScore > 70 && risk < userTolerance
  };
}
```

```typescript
// packages/arbitrage-engine/src/index.ts

export class ArbitrageEngine {
  private scouts: OpportunityScout[];
  private analyzer: OpportunityAnalyzer;
  private executor: TradeExecutor;
  private riskManager: RiskManager;

  async findOpportunities(): Promise<Opportunity[]> {
    // Run all scouts in parallel
    const opportunities = await Promise.all(
      this.scouts.map(scout => scout.scan())
    );

    return opportunities.flat();
  }

  async analyzeOpportunity(opp: Opportunity): Promise<Analysis> {
    // Score opportunity
    const score = this.analyzer.score(opp);

    // Check risk limits
    const riskCheck = this.riskManager.assess(opp, userBudget);

    return { score, riskCheck, shouldExecute: score > 70 && riskCheck.approved };
  }

  async executeOpportunity(opp: Opportunity): Promise<Result> {
    // Purchase
    const purchase = await this.executor.buy(opp);

    // List for resale
    const listing = await this.executor.sell(opp);

    // Track
    await this.trackResult(purchase, listing);

    return { success: true, estimatedProfit: opp.expectedProfit };
  }
}
```
